Model User

name       'string'
email      'string'
telephone  'integer'
id         'integer primary key'
created_at 'datetime'
updated_at 'datetime'

has_and_belongs_to_many :events
had_many :invites

join_table

create_join_table :users, :events do |t|
  t.index :user_id
  t.index :event_id
end

Model Event

location    'string'
description 'string
starts      'datetime'
ends        'datetime'
id          'integer primary key'
created_at  'datetime'
updated_at  'datetime'

has_and_belongs_to_many :users
has_many :invites

Model Invite

event      'event_id'
creator    'user_id'
id         'integer primary key'
created_at 'datetime'
updated_at 'datetime'

belongs_to :creator, foreign_key :user_id, class_name: 'user'
belongs_to :event
